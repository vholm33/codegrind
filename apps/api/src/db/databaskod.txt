CREATE TABLE railway.problems (
    id INT AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(100) NOT NULL,
    description TEXT NOT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE railway.tests (
    id INT AUTO_INCREMENT PRIMARY KEY,
    problem_id INT NOT NULL,
    input TEXT NOT NULL,
    expected TEXT NOT NULL,
    is_hidden BOOLEAN NOT NULL DEFAULT TRUE,
    FOREIGN KEY (problem_id) REFERENCES railway.problems(id)
);

create table railway.users (
    id int auto_increment primary key,
    username varchar(50) not null,
    email varchar(50) not null,
    password_hash varchar(100) not null,
    created_at datetime(3) default CURRENT_TIMESTAMP(3) null
);

//Loggar användarens submission och resultat
CREATE TABLE railway.submissions (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    problem_id INT NOT NULL,
    code TEXT NOT NULL,
    passed BOOLEAN NOT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (problem_id) REFERENCES railway.problems(id)
);
